162 {

165  std::future<std::shared_ptr<ros2_canopen::LelyMotionControllerBridge>> f = prom->get_future();
166  this->exec_->post(
167  [this, prom]()
168  {
169  std::scoped_lock<std::mutex> lock(this->driver_mutex_);
170  mc_driver_ = std::make_shared<LelyMotionControllerBridge>(*(this->exec_), *(this->master_), this->node_id_, this->node_->get_name());
171  mc_driver_->Boot();
172  prom->set_value(mc_driver_);
173  });
174  auto future_status = f.wait_for(this->non_transmit_timeout_);
175  if (future_status != std::future_status::ready)
176  {

178  throw DriverException("add_to_master: Adding timed out.");
179  }
180  this->mc_driver_ = f.get();
181  this->motor_ = std::make_shared<Motor402>(mc_driver_);
182  this->lely_driver_ = std::static_pointer_cast<LelyDriverBridge>(mc_driver_);

184  if (!this->mc_driver_->IsReady())

186  RCLCPP_WARN(this->node_->get_logger(), "Wait for device to boot.");
187  try
